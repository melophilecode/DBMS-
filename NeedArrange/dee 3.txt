create table emp(id integer,name varchar(20),age integer,salary integer);
insert into emp values(1,'varun',23,35000);
insert into emp values(2,'thara',28,41000);
insert into emp values(3,'chinu',27,31000);
select * from emp;
select * from emp where salary=(select min(salary)from emp);
select e.id,e.salary from emp e where salary in(select min(e.salary) from emp e group by e.id);
select e.id,e.age,e.salary from emp e where(e.age,e.salary) in(select e.age,e.salary from emp where e.id=3);
select a.name,b.age from emp a,(select id,avg(age)as age from emp group by id)b where b.id=a.id;
select id,name,(select count(age) as age from emp a where a.id=b.id group by id)sameage from emp b;
